230     class ChangePanelListener implements ActionListener {
231         
232         JPanel panel;
233 
234         public ChangePanelListener(JPanel panel) {
235             this.panel = panel;
236         }
237 
238         @Override
239         public void actionPerformed(ActionEvent e) {
240             Utils.saveConfiguration();
241             contentPane.removeAll();
242             contentPane.add(Box.createVerticalGlue());
243             contentPane.add(panel);
244             LOGGER.info("Panel changed: " + panel.getClass().getSimpleName());
245             contentPane.add(Box.createVerticalGlue());
246             contentPane.revalidate();
247             contentPane.repaint();
248         }
249         
250     }
