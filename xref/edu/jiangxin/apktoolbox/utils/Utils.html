36      public static void setJFrameCenterInScreen(JFrame frame) {
37          if (frame == null) {
38              logger.error("frame is null");
39              return;
40          }
41          int windowWidth = frame.getWidth();
42          int windowHeight = frame.getHeight();
43          Toolkit kit = Toolkit.getDefaultToolkit();
44          if (kit == null) {
45              logger.error("kit is null");
46              return;
47          }
48          Dimension screenSize = kit.getScreenSize();
49          if (screenSize == null) {
50              logger.error("screenSize is null");
51              return;
52          }
53          int screenWidth = screenSize.width;
54          int screenHeight = screenSize.height;
55          frame.setLocation(screenWidth / 2 - windowWidth / 2, screenHeight / 2 - windowHeight / 2);
56      }

58      public static String loadStream(InputStream in) {
59          StringBuffer buffer = new StringBuffer();
60          if (in == null) {
61              logger.error("in is null");
62              return buffer.toString();
63          }
64  
65          BufferedInputStream bis = new BufferedInputStream(in);
66          try {
67              int ptr = 0;
68              while ((ptr = bis.read()) != -1) {
69                  buffer.append((char) ptr);
70              }
71          } catch (IOException e) {
72              logger.error("read bis error", e);
73          } finally {
74              if (bis != null) {
75                  try {
76                      bis.close();
77                  } catch (IOException e) {
78                      logger.error("bis close error", e);
79                  }
80              }
81              try {
82                  in.close();
83              } catch (IOException e) {
84                  logger.error("in close error", e);
85              }
86          }
87          return buffer.toString();
88      }

100     public static Configuration getConfiguration() {
101         if (builder == null) {
102             File confiFile = new File(System.getenv("USERPROFILE") + File.separator + "apktoolboxgui.properties");
103             if (!confiFile.exists()) {
104                 try {
105                     logger.info("confiFile does not exist");
106                     boolean ret = confiFile.createNewFile();
107                     if (!ret) {
108                         logger.error("createNewFile fail");
109                     }
110                 } catch (IOException e) {
111                     logger.error("createNewFile fail", e);
112                 }
113             }
114             logger.info("builder is null, create it");
115             Parameters params = new Parameters();
116             builder = new FileBasedConfigurationBuilder<FileBasedConfiguration>(PropertiesConfiguration.class)
117                     .configure(params.properties().setFileName("apktoolboxgui.properties"));
118         }
119         Configuration conf = null;
120         try {
121             conf = builder.getConfiguration();
122         } catch (ConfigurationException e) {
123             logger.error("getConfiguration error", e);
124         }
125         return conf;
126     }

141     public static String getToolsPath() {
142         String tmp = Utils.class.getProtectionDomain().getCodeSource().getLocation().getPath();
143         if (StringUtils.isEmpty(tmp)) {
144             logger.error("tmp is empty");
145             return null;
146         }
147         File file = new File(tmp);
148         if (!file.exists()) {
149             logger.error("file does not exist");
150             return null;
151         }
152         while (file.getParentFile().exists()) {
153             File parent = file.getParentFile();
154             File tools = new File(parent, "tools");
155             if (tools.exists()) {
156                 try {
157                     return tools.getCanonicalPath();
158                 } catch (IOException e) {
159                     logger.error("getCanonicalPath fail");
160                     return null;
161                 }
162             }
163             file = file.getParentFile();
164         }
165 
166         return null;
167 
168     }

178         while(container != null) {

183         }

198     public static void blockedExecutor(String cmd) {
199         try {
200             logger.info(cmd);
201             Process process = Runtime.getRuntime().exec(cmd);
202             new StreamHandler(process.getInputStream(), 0).start();
203             new StreamHandler(process.getErrorStream(), 1).start();
204             process.waitFor();
205             logger.info("cmd finish");
206         } catch (IOException e1) {
207             logger.error("cmd fail", e1);
208         } catch (InterruptedException e1) {
209             logger.error("cmd fail", e1);
210             Thread.currentThread().interrupt();
211         }
212     }
