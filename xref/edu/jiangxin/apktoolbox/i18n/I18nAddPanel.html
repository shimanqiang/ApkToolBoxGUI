121     private void createOperationPanel() {
122         operationPanel = new JPanel();
123         Utils.setJComponentSize(operationPanel, PANEL_WIDTH, CHILD_PANEL_HIGHT);
124         operationPanel.setLayout(new BoxLayout(operationPanel, BoxLayout.X_AXIS));
125         
126         addButton = new JButton(bundle.getString("i18n.add.title"));
127         Utils.setJComponentSize(addButton, CHILD_PANEL_RIGHT_WIDTH, CHILD_PANEL_HIGHT);
128         addButton.addMouseListener(new MouseAdapter() {
129             @Override
130             public void mousePressed(MouseEvent e) {
131                 super.mousePressed(e);
132                 File srcFile = new File(srcTextField.getText());
133                 if (!srcFile.exists() || !srcFile.isDirectory()) {
134                     logger.error("srcFile is invalid");
135                     Toolkit.getDefaultToolkit().beep();
136                     JOptionPane.showMessageDialog(I18nAddPanel.this, "Source directory is invalid", "ERROR",
137                             JOptionPane.ERROR_MESSAGE);
138                     srcTextField.requestFocus();
139                     return;
140                 }
141                 String srcPath;
142                 try {
143                     srcPath = srcFile.getCanonicalPath();
144                 } catch (IOException e2) {
145                     logger.error("getCanonicalPath fail");
146                     return;
147                 }
148                 conf.setProperty("i18n.add.src.dir", srcPath);
149 
150                 List<String> targetPaths = new ArrayList<>();
151                 String[] tmps = targetTextField.getText().split(";");
152                 for (String tmp : tmps) {
153                     File targetFile = new File(tmp);
154                     if (!targetFile.exists() || !targetFile.isDirectory()) {
155                         logger.error("targetFile is invalid");
156                         Toolkit.getDefaultToolkit().beep();
157                         JOptionPane.showMessageDialog(I18nAddPanel.this, "Target directory is invalid", "ERROR",
158                                 JOptionPane.ERROR_MESSAGE);
159                         targetTextField.requestFocus();
160                         return;
161                     }
162                     try {
163                         targetPaths.add(targetFile.getCanonicalPath());
164                     } catch (IOException e1) {
165                         logger.error("getCanonicalPath fail");
166                         return;
167                     }
168                 }
169                 conf.setProperty("i18n.add.target.dir", targetTextField.getText());
170 
171                 List<String> items = new ArrayList<>();
172                 tmps = itemTextField.getText().split(";");
173                 for (String tmp : tmps) {
174                     items.add(tmp);
175                 }
176                 conf.setProperty("i18n.add.items", itemTextField.getText());
177 
178                 for (String targetPath : targetPaths) {
179                     for (String item : items) {
180                         int ret = innerProcessor(srcPath, targetPath, item);
181                         if (ret != 0) {
182                             Toolkit.getDefaultToolkit().beep();
183                             JOptionPane.showMessageDialog(I18nAddPanel.this, "Failed, please see the log", "ERROR",
184                                     JOptionPane.ERROR_MESSAGE);
185                             return;
186                         }
187                     }
188                 }
189             }
190         });
191 
192         operationPanel.add(addButton);
193     }

285     private int innerProcessor(String sourceBaseStr, String targetBaseStr, String itemName) {
286         if (StringUtils.isAnyEmpty(sourceBaseStr, targetBaseStr, itemName)) {
287             logger.error("params are invalid: sourceBaseStr: " + sourceBaseStr + ", targetBaseStr: " + targetBaseStr
288                     + ", itemName: " + itemName);
289             return -1;
290         }
291         File sourceBaseFile = new File(sourceBaseStr);
292         File targetBaseFile = new File(targetBaseStr);
293         int count = 0;
294 
295         File[] sourceParentFiles = sourceBaseFile.listFiles(new FileFilter() {
296             @Override
297             public boolean accept(File pathname) {
298                 return pathname.getName().startsWith("values");
299             }
300         });
301         if (sourceParentFiles == null) {
302             logger.error("sourceParentFiles is null");
303             return -1;
304         }
305         for (File sourceParentFile : sourceParentFiles) {
306             File sourceFile = new File(sourceParentFile, "strings.xml");
307 
308             Element sourceElement = getSourceElement(sourceFile, itemName);
309             if (sourceElement == null) {
310                 logger.warn("sourceElement is null: " + sourceFile);
311                 continue;
312             }
313 
314             File targetFile = new File(new File(targetBaseFile, sourceParentFile.getName()), "strings.xml");
315             if (!targetFile.exists()) {
316                 logger.warn("targetFile does not exist: " + sourceFile);
317                 continue;
318             }
319             try {
320                 prePocess(targetFile);
321             } catch (IOException e) {
322                 logger.error("prePocess failed.", e);
323                 return -1;
324             }
325             boolean res = setTargetElement(targetFile, sourceElement, itemName);
326             if (!res) {
327                 logger.error("setTargetElement failed.");
328                 return -1;
329             }
330             try {
331                 postProcess(targetFile);
332             } catch (IOException e) {
333                 logger.error("postProcess failed.", e);
334                 return -1;
335             }
336             logger.info("count: " + (++count) + ", in path: " + sourceFile + ", out path: " + targetFile);
337         }
338         logger.info("finish one cycle");
339         return 0;
340     }

342     private Element getSourceElement(File sourceFile, String itemName) {
343         if (!sourceFile.exists()) {
344             logger.warn("sourceFile does not exist: " + sourceFile);
345             return null;
346         }
347         SAXBuilder builder = new SAXBuilder();
348         Document sourceDoc = null;
349         try (InputStream in = new FileInputStream(sourceFile)) {
350             sourceDoc = builder.build(in);
351             logger.info("build source document: " + sourceFile);
352         } catch (JDOMException | IOException e) {
353             logger.error("build source document failed: " + sourceFile, e);
354             return null;
355         }
356         if (sourceDoc == null) {
357             logger.error("sourceDoc is null");
358             return null;
359         }
360         Element sourceElement = null;
361         for (Element sourceChild : sourceDoc.getRootElement().getChildren()) {
362             String sourceValue = sourceChild.getAttributeValue("name");
363             if (sourceValue != null && sourceValue.equals(itemName)) {
364                 sourceElement = sourceChild.clone();
365                 break;
366             }
367         }
368         return sourceElement;
369     }

371     private boolean setTargetElement(File targetFile, Element sourceElement, String itemName) {
372         SAXBuilder builder = new SAXBuilder();
373         Document targetDoc;
374         try {
375             targetDoc = builder.build(targetFile);
376             logger.info("build target document: " + targetFile);
377         } catch (JDOMException | IOException e) {
378             logger.error("build target document failed: " + targetFile, e);
379             return false;
380         }
381         Element targetRoot = targetDoc.getRootElement();
382         boolean isFinished = false;
383         for (Element targetChild : targetRoot.getChildren()) {
384             String targetValue = targetChild.getAttributeValue("name");
385             if (targetValue != null && targetValue.equals(itemName)) {
386                 targetChild.setText(sourceElement.getText());
387                 isFinished = true;
388                 break;
389             }
390         }
391         if (!isFinished) {
392             targetRoot.addContent("    ");
393             targetRoot.addContent(sourceElement);
394             targetRoot.addContent("\n");
395         }
396         XMLOutputter out = new XMLOutputter();
397         Format format = Format.getRawFormat();
398         format.setEncoding("UTF-8");
399         format.setLineSeparator("\n");
400         out.setFormat(format);
401         OutputStream os = null;
402         try {
403             os = new FileOutputStream(targetFile);
404             out.output(targetDoc, os);
405         } catch (IOException e) {
406             logger.error("output fail", e);
407             return false;
408         } finally {
409             if (os != null) {
410                 try {
411                     os.close();
412                 } catch (IOException e) {
413                     logger.error("close output stream exception", e);
414                 }
415             }
416         }
417         return true;
418     }

420     private static void prePocess(File file) throws IOException {
421         String content = FileUtils.readFileToString(file, CHARSET);
422         for (Map.Entry<String, String> entry : replace.entrySet()) {
423             content = content.replaceAll(entry.getKey(), entry.getValue());
424         }
425         FileUtils.writeStringToFile(file, content, CHARSET);
426     }
427 
428     private static void postProcess(File file) throws IOException {
429         String content = FileUtils.readFileToString(file, CHARSET);
430         for (Map.Entry<String, String> entry : replace.entrySet()) {
431             content = content.replaceAll(entry.getValue(), entry.getKey());
432         }
433         if (REMOVE_LAST_LF_OPEN) {
434             content = StringUtils.removeEnd(content, "\n");
435         }
436         FileUtils.writeStringToFile(file, content, CHARSET);
437     }
