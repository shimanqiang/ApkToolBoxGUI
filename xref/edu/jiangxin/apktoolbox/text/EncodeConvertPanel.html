41  public class EncodeConvertPanel extends EasyPanel {
42      private static final long serialVersionUID = 1L;
43  
44      private static final int CONTENT_WIDTH = 500;
45  
46      public EncodeConvertPanel() throws HeadlessException {
47          super();
48  
49          setLayout(new BoxLayout(this, BoxLayout.Y_AXIS));
50  
51          JPanel sourcePanel = new JPanel();
52          sourcePanel.setPreferredSize(new Dimension(CONTENT_WIDTH, Constants.BUTTON_HIGHT));
53          sourcePanel.setLayout(new BoxLayout(sourcePanel, BoxLayout.X_AXIS));
54          add(sourcePanel);
55          add(Box.createVerticalStrut(10));
56  
57          JPanel optionPanel = new JPanel();
58          optionPanel.setPreferredSize(new Dimension(CONTENT_WIDTH, Constants.BUTTON_HIGHT));
59          optionPanel.setLayout(new BoxLayout(optionPanel, BoxLayout.X_AXIS));
60          add(optionPanel);
61          add(Box.createVerticalStrut(10));
62  
63          JPanel operationPanel = new JPanel();
64          operationPanel.setPreferredSize(new Dimension(CONTENT_WIDTH, Constants.BUTTON_HIGHT));
65          operationPanel.setLayout(new BoxLayout(operationPanel, BoxLayout.X_AXIS));
66          add(operationPanel);
67  
68          String lastPath = conf.getString("encodeconvert.src");
69          JTextField srcTextField = new JTextField();
70          if (StringUtils.isNotEmpty(lastPath)) {
71              srcTextField.setText(lastPath);
72          }
73  
74          JButton srcButton = new JButton("File/Directory");
75          srcButton.addMouseListener(new MouseAdapter() {
76              @Override
77              public void mousePressed(MouseEvent e) {
78                  super.mousePressed(e);
79                  JFileChooser jfc = new JFileChooser();
80                  jfc.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
81                  jfc.setDialogTitle("select file or directory");
82                  if (StringUtils.isNotEmpty(lastPath)) {
83                      File lastFile = new File(lastPath);
84                      if (lastFile.exists()) {
85                          jfc.setSelectedFile(lastFile);
86                      }
87                  }
88                  int ret = jfc.showDialog(new JLabel(), null);
89                  switch (ret) {
90                  case JFileChooser.APPROVE_OPTION:
91                      File file = jfc.getSelectedFile();
92                      srcTextField.setText(file.getAbsolutePath());
93                      break;
94                  default:
95                      break;
96                  }
97  
98              }
99          });
100 
101         sourcePanel.add(srcTextField);
102         sourcePanel.add(Box.createHorizontalStrut(5));
103         sourcePanel.add(srcButton);
104         srcTextField.setAlignmentY(Component.CENTER_ALIGNMENT);
105         srcButton.setAlignmentY(Component.CENTER_ALIGNMENT);
106 
107         JLabel suffixLabel = new JLabel("Suffix:");
108         JTextField suffixTextField = new JTextField();
109         suffixTextField.setText(conf.getString("encodeconvert.suffix"));
110         suffixTextField.setPreferredSize(new Dimension(60, Constants.BUTTON_HIGHT));
111         suffixTextField.setMaximumSize(new Dimension(60, Constants.BUTTON_HIGHT));
112 
113         Set<String> charsets = Charset.availableCharsets().keySet();
114 
115         JLabel fromLabel = new JLabel("From:");
116         AutoCompleteComboBox<String> fromComboBox = new AutoCompleteComboBox<String>();
117         fromComboBox.setPreferredSize(new Dimension(80, Constants.BUTTON_HIGHT));
118         fromComboBox.setEnabled(false);
119 
120         JCheckBox autoDetectCheckBox = new JCheckBox("Auto Detect");
121         autoDetectCheckBox.setSelected(true);
122         autoDetectCheckBox.addItemListener(new ItemListener() {
123             @Override
124             public void itemStateChanged(ItemEvent e) {
125                 if (e.getStateChange() == ItemEvent.SELECTED) {
126                     fromComboBox.setEnabled(false);
127                 } else {
128                     fromComboBox.setEnabled(true);
129                 }
130 
131             }
132         });
133 
134         JLabel toLabel = new JLabel("To:");
135         AutoCompleteComboBox<String> toComboBox = new AutoCompleteComboBox<String>();
136         toComboBox.setPreferredSize(new Dimension(80, Constants.BUTTON_HIGHT));
137         toComboBox.setSelectedItem(conf.getString("encodeconvert.to"));
138 
139         for (String charset : charsets) {
140             fromComboBox.addItem(charset);
141             toComboBox.addItem(charset);
142         }
143 
144         optionPanel.add(suffixLabel);
145         optionPanel.add(Box.createHorizontalStrut(5));
146         optionPanel.add(suffixTextField);
147         optionPanel.add(Box.createHorizontalGlue());
148         optionPanel.add(fromLabel);
149         optionPanel.add(Box.createHorizontalStrut(5));
150         optionPanel.add(fromComboBox);
151         optionPanel.add(Box.createHorizontalGlue());
152         optionPanel.add(autoDetectCheckBox);
153         optionPanel.add(Box.createHorizontalGlue());
154         optionPanel.add(toLabel);
155         optionPanel.add(Box.createHorizontalStrut(5));
156         optionPanel.add(toComboBox);
157 
158         JButton sceenshotButton = new JButton("Convert");
159         sceenshotButton.addMouseListener(new MouseAdapter() {
160             @Override
161             public void mousePressed(MouseEvent e) {
162                 super.mousePressed(e);
163                 File srcFile = new File(srcTextField.getText());
164                 if (!srcFile.exists()) {
165                     logger.error("srcFile is invalid");
166                     Toolkit.getDefaultToolkit().beep();
167                     JOptionPane.showMessageDialog(EncodeConvertPanel.this, "Source file is invalid", "ERROR",
168                             JOptionPane.ERROR_MESSAGE);
169                     srcTextField.requestFocus();
170                     return;
171                 }
172                 String srcPath;
173                 try {
174                     srcPath = srcFile.getCanonicalPath();
175                 } catch (IOException e2) {
176                     logger.error("getCanonicalPath fail");
177                     return;
178                 }
179                 conf.setProperty("encodeconvert.src", srcPath);
180                 conf.setProperty("encodeconvert.suffix", suffixTextField.getText());
181                 conf.setProperty("encodeconvert.to", toComboBox.getSelectedItem().toString());
182 
183                 try {
184                     ArrayList<File> files = new ArrayList<File>();
185                     files.addAll(new FileFilterWrapper().list(srcPath, suffixTextField.getText()));
186                     for (File file : files) {
187                         logger.info("process: " + file.getCanonicalPath());
188                         String fromEncoder;
189                         if (autoDetectCheckBox.isSelected()) {
190                             fromEncoder = EncoderDetector.judgeFile(file.getCanonicalPath());
191                         } else {
192                             fromEncoder = fromComboBox.getSelectedItem().toString();
193                         }
194                         logger.info("from: " + fromEncoder);
195                         String toEncoder = toComboBox.getSelectedItem().toString();
196                         EncoderConvert.encodeFile(file.getCanonicalPath(), fromEncoder, toEncoder);
197                     }
198 
199                     logger.info("convert finish");
200                 } catch (IOException e1) {
201                     logger.error("convert fail", e1);
202                 }
203             }
204         });
205 
206         operationPanel.add(sceenshotButton);
207     }
