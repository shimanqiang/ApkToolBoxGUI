22      public static boolean isTraditional(String in) {
23  
24          ZhConverter converter = ZhConverter.getInstance(ZhConverter.SIMPLIFIED);
25          Properties dict = converter.getDict();
26          for (int i = 0; i < in.length(); i++) {
27              char c = in.charAt(i);
28              String key = "" + c;
29              if (dict.containsKey(key)) {
30                  return true;
31              }
32          }
33          return false;
34      }
35  
36      public static boolean isContainsChinese(String str) {
37  
38          String regEx = "[\u4e00-\u9fa5]";
39          Pattern pat = Pattern.compile(regEx);
40          Matcher matcher = pat.matcher(str);
41          boolean flg = false;
42          if (matcher.find()) {
43              flg = true;
44          }
45          return flg;
46      }
47  
48      public static boolean isAllChinese(String str) {
49  
50          boolean flag = true;
51          if (str == null || str.trim().isEmpty()) {
52              return false;
53          }
54  
55          int length = str.length();
56          for (int i = 0; i < length; i++) {
57              if (!isContainsChinese(str.substring(i, i + 1))) {
58                  flag = false;
59                  break;
60              }
61          }
62  
63          return flag;
64      }
