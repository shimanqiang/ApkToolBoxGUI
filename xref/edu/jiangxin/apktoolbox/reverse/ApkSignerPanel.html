270     private final class ScreenshotButtonMouseAdapter extends MouseAdapter {
271         private File apkFile;
272         private String apkPath;
273         private File keystoreFile;
274         private String keystorePath;
275         private String keystorePassword;
276         private String alias;
277         private String alisaPassword;
278 
279         @Override
280         public void mousePressed(MouseEvent e) {
281             super.mousePressed(e);
282             apkFile = new File(apkPathTextField.getText());
283             if (!apkFile.exists()) {
284                 logger.error("srcFile is invalid");
285                 Toolkit.getDefaultToolkit().beep();
286                 JOptionPane.showMessageDialog(ApkSignerPanel.this, "apk file is invalid", "ERROR",
287                         JOptionPane.ERROR_MESSAGE);
288                 apkPathTextField.requestFocus();
289                 return;
290             }
291 
292             keystoreFile = new File(keyStorePathTextField.getText());
293             if (!keystoreFile.exists()) {
294                 logger.error("keystoreFile is invalid");
295                 Toolkit.getDefaultToolkit().beep();
296                 JOptionPane.showMessageDialog(ApkSignerPanel.this, "keystore file is invalid", "ERROR",
297                         JOptionPane.ERROR_MESSAGE);
298                 keyStorePathTextField.requestFocus();
299                 return;
300             }
301             
302             try {
303                 apkPath = apkFile.getCanonicalPath();
304                 keystorePath = keystoreFile.getCanonicalPath();
305             } catch (IOException e2) {
306                 logger.error("getCanonicalPath fail");
307                 return;
308             }
309             
310             conf.setProperty("apksigner.apk.path", apkPath);
311             conf.setProperty("apksigner.keystore.path", keystorePath);
312 
313             keystorePassword = keyStorePasswordField.getText();
314             if (StringUtils.isEmpty(keystorePassword)) {
315                 logger.error("keystorePassword is invalid");
316                 Toolkit.getDefaultToolkit().beep();
317                 JOptionPane.showMessageDialog(ApkSignerPanel.this, "keystorePassword is invalid", "ERROR",
318                         JOptionPane.ERROR_MESSAGE);
319                 keyStorePasswordField.requestFocus();
320                 return;
321             }
322             conf.setProperty("apksigner.keystore.password", keystorePassword);
323 
324             alias = aliasTextField.getText();
325             if (StringUtils.isEmpty(alias)) {
326                 logger.error("alias is invalid");
327                 Toolkit.getDefaultToolkit().beep();
328                 JOptionPane.showMessageDialog(ApkSignerPanel.this, "alias is invalid", "ERROR",
329                         JOptionPane.ERROR_MESSAGE);
330                 aliasTextField.requestFocus();
331                 return;
332             }
333             conf.setProperty("apksigner.alias", alias);
334 
335             alisaPassword = aliasPasswordField.getText();
336             if (StringUtils.isEmpty(alisaPassword)) {
337                 logger.error("alisaPassword is invalid");
338                 Toolkit.getDefaultToolkit().beep();
339                 JOptionPane.showMessageDialog(ApkSignerPanel.this, "alisaPassword is invalid", "ERROR",
340                         JOptionPane.ERROR_MESSAGE);
341                 aliasPasswordField.requestFocus();
342                 return;
343             }
344             conf.setProperty("apksigner.alias.password", alisaPassword);
345 
346             Utils.blockedExecutor(getCmd());
347         }
348 
349         private String getCmd() {
350             StringBuilder sb = new StringBuilder();
351             sb.append("java -jar \"-Duser.language=en\" \"-Dfile.encoding=UTF8\"").append(" \"")
352                     .append(Utils.getToolsPath()).append(File.separator).append("apksigner.jar\"")
353                     .append(" -keystore ").append(keystorePath).append(" -pswd ").append(keystorePassword)
354                     .append(" -alias ").append(alias).append(" -aliaspswd ").append(alisaPassword).append(" ")
355                     .append(apkPath);
356             return sb.toString();
357         }
